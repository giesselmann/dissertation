# BASE STAGE
FROM ubuntu:18.04 as base_stage
RUN apt-get --yes update && apt-get install -y --no-install-recommends wget

# BUILD STAGE
FROM base_stage as build_stage
## build packages
RUN apt-get --yes update && apt-get install -y --no-install-recommends gcc python3-dev python3-pip

# PACKAGE STAGE
FROM base_stage
MAINTAINER Pay Giesselmann <giesselmann@molgen.mpg.de>
## runtime packages
RUN apt-get --yes update && apt-get install -y --no-install-recommends python3 python3-pip
## copy binaries from build stage
COPY --from=build_stage /build/bin/* /usr/bin/
